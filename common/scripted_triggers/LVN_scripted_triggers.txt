LVN_we_beat_this_bad_boy = {
	custom_trigger_tooltip = {
		tooltip = LVN_we_beat_this_bad_boy
		hidden_trigger = {
			OR = {
				is_subject_of = ROOT
				all_of_scopes = {
					array = core_states
					OR = {
						is_controlled_by_ROOT_or_subject = YES
						CONTROLLER = { is_in_faction_with = ROOT }
					}
				}
			}
		}
	}
}

			}

	}
}

LVN_no_ghosts = {
	custom_trigger_tooltip = {
		tooltip = LVN_no_ghosts
		hidden_trigger = {
			all_of_scopes = { #If all of Libya cores are our cores already...
				array = ROOT.core_states
				CONTROLLER = {
					NOT = { tag = RSI }
					NOT = { tag = PCI }
				}
			}
		}
	}
}

		}
	}
}
