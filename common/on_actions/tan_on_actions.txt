on_actions = {

	on_new_term_election = {
		effect = {
			if = {
				limit = {
					tag = TAN
					has_government = democratic
				}
				country_event = tan.54
			}
			if = {
				limit = {
					tag = TAN
					has_government = communism
				}
				country_event = tan.90
			}
		}
	}
	
	on_civil_war_end = {
		effect = {
			if = {
				limit = {
					tag = TAN
					has_government = democratic
				}
				complete_national_focus = TAN_victory_in_civil_war
				delete_unit_template_and_units = {
					division_template = "Partizanskaya Diviziya"
					disband = yes
				}
			}
			if = {
				limit = {
					tag = TAN
					has_government = neutrality
				}
				complete_national_focus = TAN_long_live_anarchy
			}
		}
	}

	on_annex = {
		effect = {
			if = {
				limit = {
					tag = TAN
					has_completed_focus = TAN_center_capture
					NOT = { country_exists = MON }
					owns_state = 330
					NOT = {
						has_cosmetic_tag = MON
						has_cosmetic_tag = MON_EMP
					}
				}
				set_cosmetic_tag = MON
				every_state = {
					limit = {
						is_core_of = MON
					}
					remove_core_of = MON
				}
			}
		}
	}
	
}
