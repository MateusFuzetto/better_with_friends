
LVN_set_armistice_variables = {	
	#Armistice variables
	if = {
		limit = {
			has_country_flag = king_preparing_to_depose_mussolini_war
		}
		set_variable = { LVN_armistice_leverage = 5 } #Initial value	
	}
	if = {
		limit = {
			enemies_naval_strength_ratio < 1
		}
		add_to_variable = { LVN_armistice_leverage = 5 } #5 bonus points if we have a stronger navy than our enemies
	}
	if = {
		limit = {
			enemies_strength_ratio < 1
		}
		add_to_variable = { LVN_armistice_leverage = 5 } #5 bonus points if our allies don't have a bigger army
	}
	if = {
		limit = {
			is_in_faction = yes
			alliance_strength_ratio > 1
		}
		add_to_variable = { LVN_armistice_leverage = 5 } #5 bonus points if you are in a faction more powerful than your enemies
	}
	if = {
		limit = {
			check_variable = { LVN_armistice_leverage < 10 }
		}
		set_variable = { LVN_armistice_leverage = 10 } #5 We will set you at base 10 points to at least guarantee independence
	}
}
